package main

import (
	"encoding/json"
	"fmt"
	"net/http"
)

type ApiResponseParse struct {
	ID    int    `json:"id"`
	Title string `json:"title"`
}

type ApiResponse struct {
	Liste []ApiResponseParse `json:"posts"`
}

func main() {
	url := "https://my-json-server.typicode.com/typicode/demo/db"

	// http.Get() fonksiyonuyla GET isteği gönderiyoruz.
	response, err := http.Get(url)
	if err != nil {
		fmt.Println("API isteği gönderilemedi:", err)
		return
	}
	defer response.Body.Close()

	// Yanıtı JSON formatından ApiResponse tipine çeviriyoruz.
	var apiResponse ApiResponse
	err = json.NewDecoder(response.Body).Decode(&apiResponse)
	if err != nil {
		fmt.Println("API not working:", err)
		return
	}

	// API yanıtındaki "message" özelliğini yazdırıyoruz.
	fmt.Println("API response:", apiResponse.Liste)

	// API yanıtındaki "message" özelliğini yazdırıyoruz.
	for _, post := range apiResponse.Liste {
		fmt.Printf("Post %d: %s\n", post.ID, post.Title)
	}
}
